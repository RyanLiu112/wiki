    为什么动态链表慢？¶

    ​ C++ new 动态分配很慢，用new生成节点很容易超时

    为什么用数组模拟链表（静态链表）？¶

    单链表（数组模拟）¶

    算法题中常用邻接表，邻接表常用于存储图和树。

    // 用相同的下标关联e和ne
int e[N]; // 用数组存储每个节点的值
int ne[N]; // 用数组存储下一个节点的指针

    826.单链表¶

    链接：https://www.acwing.com/problem/content/828/

    #include <iostream>

using namespace std;

const int N = 100010;

// head 表示头节点的下标
// e[i] 表示节点i的值
// ne[i] 表示节点i的next指针
// idx 表示当前已经用到了哪个点
int head, e[N], ne[N], idx;

// 初始化
void init() {
    head = -1;
    idx = 0;
}

// 将x插到头节点
void add_to_head(int x) {
    e[idx] = x; // 储存x
    ne[idx] = head; // 将x的next指向head的下一个节点
    head = idx++; // 将head指向x，并更新idx
}

// 将x插到下标为k的节点后面
void add(int k, int x) {
    e[idx] = x; // 储存x
    ne[idx] = ne[k]; // 将x的next指向k的下一个节点
    ne[k] = idx++; // 将k指向x，并更新idx
}

// 删除下标为k的后面的节点
void remove(int k) {
    ne[k] = ne[ne[k]];
}

int main() {
    int m;
    scanf("%d", &m);

    init();

    while (m--) {
        char op[2];
        int k, x;

        scanf("%s", op);
        if (op[0] == 'H') {
            scanf("%d", &x);
            add_to_head(x);
        } else if (op[0] == 'D') {
            scanf("%d", &k);
            if (!k) head = ne[head];
            remove(k - 1);
        } else {
            scanf("%d%d", &k, &x);
            add(k - 1, x);
        }
    }
    for (int i = head; i != -1; i = ne[i]) printf("%d ", e[i]);
    printf("\n");

    return 0;
}

    易错：if (!k) head = ne[head];

    双链表（数组模拟）¶

    双链表常用于优化某些问题。

    双链表同时指向前后。

    // 双链表需要使用两个数组储存指针
int l[N]; // 每个节点指向的前一个节点
int r[N]; // 每个节点指向的后一个节点
// 双链表使用下标为0的点储存头节点（head）
// 双链表使用下标为1的点储存尾节点（tail）

    827.双链表¶

    链接：https://www.acwing.com/problem/content/829/

    #include <iostream>
#include <string.h>

using namespace std;

const int N = 100010;

// e[i] 表示节点i的值
// l[i] 表示节点i指向的前一个节点
// r[i] 表示节点i指向的后一个节点
// idx 表示当前已经用到了哪个点
int e[N], l[N], r[N], idx;

// 初始化
void init() {
    r[0] = 1, l[1] = 0; // 头节点（head）的下标为0，尾节点（tail）的下标为1
    idx = 2;
}

// 将x插到下标为k的节点后面
// 在k的前面插入可以写成 add(l[k], x)
void add(int k, int x) {
    e[idx] = x;
    l[idx] = k, r[idx] = r[k];
    l[r[k]] = idx, r[k] = idx;
    idx++;
}

// 删除下标为k的节点
void remove(int k) {
    l[r[k]] = l[k];
    r[l[k]] = r[k];
}

int main() {
    int m;
    scanf("%d", &m);

    init();

    while (m--) {
        char op[5];
        int k, x;

        scanf("%s", op);
        if (op[0] == 'L') {
            scanf("%d", &x);
            add(0, x);
        } else if (op[0] == 'R') {
            scanf("%d", &x);
            add(l[1], x);
        } else if (op[0] == 'D') {
            scanf("%d", &k);
            remove(k + 1);
        } else if (op[0] == 'I' && op[1] == 'L') {
            scanf("%d%d", &k, &x);
            add(l[k + 1], x);
        } else {
            scanf("%d%d", &k, &x);
            add(k + 1, x);
        }
    }
    for (int i = r[0]; i != 1; i = r[i]) printf("%d ", e[i]);
    printf("\n");

    return 0;
}

    易错：k+1

    栈（数组模拟）¶

    先进后出（First-In-Last-OUT，FILO）

    #include <iostream>

using namaspace std;

const int N = 100010;

int stk[N], tt = -1; // tt为栈顶下标，y总用的 tt 从 0 开始，但很多书上都是从 -1 开始

// 插入
stk[++tt] = x;

// 弹出
tt--;

// 判断栈是否为空
if (tt > 0) not empty;
else empty;

// 取出栈顶元素
stk[tt];

    单调栈¶

    单调栈题目类型不多，最常见的是：

    给定一个序列，求每个数左边离它最近的且比它小的数。

    类似双指针的做法，先想暴力做法，然后挖掘性质简化问题

    若 \(a_i \ge a_j\) 且 \(i < j\)，则可以删去 \(a_i\)。

    故最后剩下的栈是一个单调栈。

    830.单调栈¶

    链接：https://www.acwing.com/problem/content/832/

    #include <iostream>

using namespace std;

const int N = 100010;

int stk[N], tt = -1;

int main() {
    int n;
    scanf("%d", &n);

    while (n--) {
        int x;
        scanf("%d", &x);

        while (tt >= 0 && stk[tt] >= x) tt--;
        if (tt >= 0) printf("%d ", stk[tt]);
        else printf("-1 ");

        stk[++tt] = x;
    }

    return 0;
}

    易错：while (tt && stk[tt] >= x) tt--;

    单调队列¶

    题型不多，最常见的是滑动窗口，窗口可以用队列维护。

    类似双指针的做法，先想暴力做法，然后挖掘性质简化问题

    若 \(a_i \ge a_j\) 且 \(i < j\)，则可以删去 \(a_i\)。

    故最后剩下的队列是一个单调队列。

    154.滑动窗口¶

    链接：https://www.acwing.com/problem/content/156/

    #include <iostream>

using namespace std;

const int N = 1000010;

int n, k;
int a[N], q[N], hh = 0, tt = -1;

int main() {
    scanf("%d%d", &n, &k);

    for (int i = 0; i < n; ++i) scanf("%d", &a[i]);

    for (int i = 0; i < n; ++i) {
        // 判断队头是否已经滑出窗口
        if (hh <= tt && i - k + 1 > q[hh]) hh++;
        while (hh <= tt && a[q[tt]] >= a[i]) tt--;
        q[++tt] = i;
        if (i >= k - 1) printf("%d ", a[q[hh]]);
    }
    printf("\n");

    hh = 0, tt = -1;
    for (int i = 0; i < n; ++i) {
        // 判断队头是否已经滑出窗口
        if (hh <= tt && i - k + 1 > q[hh]) hh++;
        while (hh <= tt && a[q[tt]] <= a[i]) tt--;
        q[++tt] = i;
        if (i >= k - 1) printf("%d ", a[q[hh]]);
    }

    return 0;
}

    KMP¶

    KMP的解题思路¶

    • 先想暴力算法怎么做

    • 设 \(S[N]\) 是比较长的串（要匹配的串），\(P[M]\) 是比较短的串（模板串）

    • ```c++ int s[N], p[M];

      // KMP下标从1开始（y总个人习惯） for (int i = 1; i <= n; ++i) { bool flag = true; for (int j = 1; j <= m; ++j) { if (s[i] != p[j]) { flag = false; break; } } } ```

    • 再想如何去优化

    • 假设现在 \(P\) 的起始位置 \(P[0]\) 在 \(S\) 的第 \(i\) 个位置 \(S[i]\)，从 \(i\) 到 \(i+k\)，\(S\) 和 \(P\) 完全匹配，但 \(S[i+k+1]\) 和 \(P[k+1]\) 不匹配。

    • 如果按照暴力算法，我们把 \(P\) 向后移动一个位置，就会浪费掉 \(S[i:i+k+1]\) 和 \(P[0:k+1]\) 完全匹配的信息。
    • 反之，我们可以利用这个信息，将 \(P\) 向后移动 个位置，

    831.KMP字符串¶

    链接：https://www.acwing.com/activity/content/problem/content/869/

    #include <iostream>

using namespace std;

const int N = 100010, M = 1000010;

int n, m;
char p[N], s[M];
int ne[N];

int main() {
    cin >> n >> p + 1 >> m >> s + 1;

    // 求 next 过程
    for (int i = 2, j = 0; i <= n; ++i) {
        while (j && p[i] != p[j + 1]) j = ne[j];
        if (p[i] == p[j + 1]) j++;
        ne[i] = j;
    }

    // KMP 匹配过程
    for (int i = 1, j = 0; i <= m; ++i) {
        while (j && s[i] != p[j + 1]) j = ne[j];
        if (s[i] == p[j + 1]) j++;
        if (j == n) { // 匹配成功
            printf("%d ", i - n);
            j = ne[j];
        }
    }

    return 0;
}

    Trie 树（字典树）¶

    Trie：高效地存储和查找字符串集合的数据结构。

    字符串要么全是大写，要么全是小写，要么全是数字，……

    储存的时候在每个单词结尾的地方做一个标记，这样在查找某一个单词的时候，既要满足这个单词存在（有一条通路），也要满足最后一个字母有标记。

    835.Trie字符串统计¶

    链接：https://www.acwing.com/problem/content/837/

    #include <iostream>

using namespace std;

const int N = 100010;

// son 表示每个节点的子节点（最多26个）
// cnt 表示以当前节点结尾的单词有多少个
// idx 表示当前用到了哪个下标（与单链表相同）
int son[N][26], cnt[N], idx; // 下标是0的点，既是根节点，又是空节点
char str[N];

void insert(char str[]) {
    int p = 0;
    for (int i = 0; str[i]; ++i) { // 字符串结尾为'\0'，str[i] 可以判断是否到达结尾
        int u = str[i] - 'a'; // 当前子节点的编号（映射到0-25）
        if (!son[p][u]) son[p][u] = ++idx;
        p = son[p][u];
    }

    cnt[p]++;
}

int query(char str[]) {
    int p = 0;
    for (int i = 0; str[i]; ++i) {
        int u = str[i] - 'a';
        if (!son[p][u]) return 0;
        p = son[p][u];
    }

    return cnt[p];
}

int main() {
    int n;
    scanf("%d", &n);
    while (n--) {
        char op[2];
        scanf("%s%s", op, str);
        if (op[0] == 'I') insert(str);
        else printf("%d\n", query(str));
    }
    return 0;
}

    if (!son[p][u]) return 0;

    143.最大异或对¶

    链接：https://www.acwing.com/problem/content/145/

    分析：根据题目 \(1 \le N \le 10^5\)，可以推算出这个题的复杂度应该是 \(\mathcal{O}(n\log n)\)。

    y总说一道题先用暴力去写，然后再考虑用什么数据结构去优化。

    暴力：

    优化思路：遍历 \(A[i]\) 不变，考虑优化遍历 \(A[j]\)，时间复杂度为 \(10^5\times 31\approx 3\times 10^6\)

    使用 Trie 树优化后：

    #include <iostream>
#include <algorithm>

using namespace std;

const int N = 100010, M = 3000000;

int son[M][2], idx;
int a[N];

void insert(int x) {
    int p = 0;
    for (int i = 30; i >= 0; --i) {
        int u = x >> i & 1; // x的第i位是0还是1
        if (!son[p][u]) son[p][u] = ++idx;
        p = son[p][u];
    }
}

int query(int x) {
    int res = 0, p = 0;
    for (int i = 30; i >= 0; --i) {
        int u = x >> i & 1; // x的第i位是0还是1
        if (son[p][!u]) {
            res += 1 << i;
            p = son[p][!u];
        } else p = son[p][u];
    }
    return res;
}

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) {
        scanf("%d", &a[i]);
        insert(a[i]);
    }

    int res = 0;
    for (int i = 0; i < n; ++i) res = max(res, query(a[i]));

    printf("%d\n", res);

    return 0;
}

    并查集¶

    面试和比赛常出的数据结构，代码短但思路巧妙。

    作用¶

    快速地（以近乎 \(\mathcal{O}(1)\) 的复杂度）处理以下操作：

    1. 将两个集合合并
    2. 查询两个元素是否在一个集合中

    基本原理¶

    以树的形式维护所有集合（不一定是二叉树），每个集合都用树来维护。

    每个集合的编号为其根节点的编号。每个点都存储它的父节点 \(p[x]\)。

    当我们想求某个节点的所属集合时，先找它的父节点，然后一层一层找到根节点。

    如何判断树根？¶

    if (p[x] == x)

    如何求 \(x\) 的集合编号¶

    while (p[x] != x) x = p[x];

    如何合并两个集合¶

    直接把其中一个树的根节点插到另一个树的某一个节点

    设 \(p[x]\) 是 \(x\) 的集合编号，\(p[y]\) 是 \(y\) 的集合编号，令 \(p[x]=y\)

    并查集的优化——路径压缩（降低树的高度）¶

    在求 \(x\) 的集合编号时，把路径上所有子节点都指向根节点。

    并查集用这个优化后，可以看作是 \(\mathcal{O}(1)\) 的复杂度。

    并查集还有一种优化是按秩合并，但优化效果不明显，没什么用。

    关键¶

    只需要记住 find 函数。

    836.合并集合¶

    链接：https://www.acwing.com/problem/content/838/

    #include <iostream>

using namespace std;

const int N = 100010;

int n, m;
int p[N]; // 储存每个元素的父节点

int find(int x) { // 返回 x 的祖宗节点（不是父节点） + 路径压缩
    if (p[x] != x) p[x] = find(p[x]);
    return p[x];
}

int main() {
    scanf("%d%d", &n, &m);

    for (int i = 1; i <= n; ++i) p[i] = i; // 每个元素自己一个集合，根节点为自己

    while (m--) {
        char op[2]; // 如果用 op 和 %c 的话，可能会读入一些空格、回车或制表符，但字符串会忽略空格、回车和制表符
        int a, b;
        scanf("%s%d%d", op, &a, &b);

        if (op[0] == 'M') p[find(a)] = find(b);
        else {
            if (find(a) == find(b)) printf("Yes\n");
            else printf("No\n");
        }
    }

    return 0;
}

    240.食物链¶

    链接：https://www.acwing.com/problem/content/242/

    维护每个点到根节点的距离，只需要知道两个点到根节点的距离，就可以判断它们之间的关系，不需要两两判断一次（\(\mathcal{O}(n^2)\)）。

    #include <iostream>

using namespace std;

const int N = 50010;

int n, k;
int p[N], d[N];

int find(int x) {
    if (p[x] != x) {
        int t = find(p[x]);
        d[x] += d[p[x]]; // 根据路径压缩，将x到根节点的距离加上x的父节点到根节点的距离
        p[x] = t; // 路径压缩
    }
    return p[x];
}

int main() {
    scanf("%d%d", &n, &k);

    for (int i = 1; i <= n; ++i) p[i] = i;

    int cnt = 0;
    while (k--) {
        int t, x, y;
        scanf("%d%d%d", &t, &x, &y);

        if (x > n || y > n) cnt++;
        else {
            int px = find(x), py = find(y);
            if (t == 1) {
                if (px == py && (d[x] - d[y]) % 3) cnt++;
                else if (px != py) {
                    p[px] = py;
                    d[px] = d[y] - d[x];
                }
            } else {
                if (px == py && (d[x] - d[y] - 1) % 3) cnt++;
                else if (px != py) {
                    p[px] = py;
                    d[px] = d[y] - d[x] + 1;
                }
            }
        }
    }

    printf("%d\n", cnt);

    return 0;
}

    堆¶

    STL 里的堆是优先队列（priority_queue）

    如何手写一个堆？

    堆：一棵完全二叉树，堆顶是最小值。除了最后一层，上面所有层都是满的，最后一层的节点从左到右排列。

    堆的存储¶

    用一个一维数组存储，下标从 1 开始。

    \(x\) 的左儿子：\(2x\)

    \(x\) 的右儿子：\(2x+1\)

    堆的操作¶

    1. 插入一个数
    2. 求集合中的最小值
    3. 删除最小值
    4. 删除任意一个元素
    5. 修改任意一个元素

    STL 的堆可以实现 1-3，但不能直接实现 4-5。

    up 和 down 的时间复杂度都是 \(\mathcal{O}(\log n)\)（和树的高度成正比）

    4、5 中的 up 和 down 不会都执行，只会执行一个，这样写是为了简便。

    838.堆排序¶

    链接：https://www.acwing.com/problem/content/840/

    #include <iostream>
#include <algorithm>

using namespace std;

const int N = 100010;

int n, m;
int h[N], idx;

void down(int u) {
    int t = u; // 三个点中最小值的编号
    if (u * 2 <= idx && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= idx && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (u != t) {
        swap(h[u], h[t]);
        down(t);
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; ++i) scanf("%d", &h[i]);
    idx = n;

    for (int i = n / 2; i; --i) down(i); // O(n) 建堆（错位相减），最后一层不需要 down

    while (m--) {
        printf("%d ", h[1]);
        h[1] = h[idx];
        idx--;
        down(1);
    }

    return 0;
}

    839.模拟堆¶

    链接：https://www.acwing.com/problem/content/841/

    用 \(ph[k]\) 储存第 \(k\) 个插入的数在堆里的下标，用 \(hp[k]\) 储存堆里第 \(k\) 个点是第几个插入的。

    关于 ph 和 hp 的解释：p 表示下标 / 指针（pointer），h 是堆（heap）的首字母，ph 表示从 p 映射到 h，hp 表示从 h 到 p 的映射。

    #include <iostream>
#include <algorithm>
#include <string.h>

using namespace std;

const int N = 100010;

int h[N], ph[N], hp[N], idx;
// ph[k] 表示第 k 个插入的数在堆里的下标
// hp[k] 表示堆里第 k 个点是第几个插入的

void heap_swap(int a, int b) {
    swap(ph[hp[a]], ph[hp[b]]);
    swap(hp[a], hp[b]);
    swap(h[a], h[b]);
}

void down(int u) {
    int t = u;
    if (u * 2 <= idx && h[u * 2] < h[t]) t = u * 2;
    if (u * 2 + 1 <= idx && h[u * 2 + 1] < h[t]) t = u * 2 + 1;
    if (u != t) {
        heap_swap(u, t);
        down(t);
    }
}

void up(int u) {
    while (u / 2 && h[u / 2] > h[u]) {
        heap_swap(u / 2, u);
        u /= 2;
    }
}

int main() {
    int n, m = 0; // m 表示第几个插入的数
    scanf("%d", &n);

    while (n--) {
        char op[5];
        int k, x;
        scanf("%s", op);

        if (!strcmp(op, "I")) {
            scanf("%d", &x);
            idx++;
            m++;
            ph[m] = idx, hp[idx] = m;
            h[idx] = x;
            up(idx);
        } else if (!strcmp(op, "PM")) printf("%d\n", h[1]);
        else if (!strcmp(op, "DM")) {
            heap_swap(1, idx);
            idx--;
            down(1);
        } else if (!strcmp(op, "D")) {
            scanf("%d", &k);
            k = ph[k];
            heap_swap(k, idx);
            idx--;
            down(k), up(k);
        } else {
            scanf("%d%d", &k, &x);
            k = ph[k];
            h[k] = x;
            down(k), up(k);
        }
    }

      return 0;
}

    哈希表¶

    哈希表¶

    把一个较大的空间映射到 \(0\sim n\)，例如将 \(-10^9 \sim 10^9 \rightarrow 0\sim 10^5\)。

    哈希函数¶

    \(\exist \ h(x)\)，\(\text{s.t.} \ \forall \ x \in [-10^9,10^9]\)，\(h(x) \in [0, 10^5]\)，则称 \(h(x)\) 为哈希函数。

    哈希函数的常用形式¶

    • mod

    \(x \ \text{mod} \ 10^5 \in [0, 10^5)\)，取模的数最好取质数，并且该质数最好离 2 的整数次幂比较远（这样取冲突的概率最小）。

    • 冲突（若干不同数映射到同一个数）

    按照冲突的处理方式，可以将哈希表分为开放寻址法和拉链法。

    时间复杂度¶

    在一般情况下，可以将哈希表的时间复杂度看作是 \(\mathcal{O}(1)\)，比如拉链法虽然每个哈希值可能对应多个数，但所有哈希值平均下来可以把每条链（单链表）看作是常数。

    存储结构¶

    • 开放寻址法（y总喜欢用）
    • 拉链法（简单）
    拉链法¶

    开一个一维数组来存储所有的哈希值，下标从 \(0\) 到 \(10^5-1\)，如果 \(h(11)=3,\ h(23)=3\)，则在 \(3\) 对应的位置添加一条链（单链表），储存 \(11\) 和 \(23\)。

    840.模拟散列表¶

    链接：https://www.acwing.com/problem/content/842/

    注：在算法题中，一般只有添加和查找操作，没有删除操作。如果一定要实现删除，也不需要真正删除，可以再开一个 bool 类型的数组，对每个点进行标记。

    拉链法¶
    #include <iostream>
#include <cstring>

using namespace std;

const int N = 100003; // 大于 100000 的第一个质数

int h[N], e[N], ne[N], idx; // h 储存的是每个单链表的头节点

void insert(int x) {
    int k = (x % N + N) % N; // k 是哈希值，在 C++ 中，负数的模是负数，因此加上 N 之后再模 N 可以保证模一定是正数。
    e[idx] = x;
    ne[idx] = h[k];
    h[k] = idx++;
}

bool find(int x) {
    int k = (x % N + N) % N;
    for (int i = h[k]; i != -1; i = ne[i])
        if (e[i] == x)
            return true;

    return false;
}

int main() {
    int n;
    scanf("%d", &n);

    memset(h, -1, sizeof h); // 把所有的槽清空，-1 表示空指针

    while (n--) {
        char op[2];
        int x;
        scanf("%s%d", op, &x);

        if (op[0] == 'I') insert(x);
        else {
            if (find(x)) printf("Yes\n");
            else printf("No\n");
        }
    }

    return 0;
}
    开放寻址法¶

    只开一个数组，不需要开链表。

    数组长度需要开到题目数据范围的 \(2\sim 3\) 倍，如题目中 \(1 \le N \le 10^5\)，则数组需要开到 \(2\times 10^5 \sim 3\times 10^5\)，可以找大于 \(2\times 10^5\) 的最小的质数。

    添加：设 \(h(x)=k\)，先看第 \(k\) 个位置是否有值，如果有值则去下一个位置继续查找，直到找到一个空位存储。

    查找：设 \(h(x)=k\)，从第 \(k\) 个位置开始向后查找，看该位置是否有值、是否等于 \(x\)，如果有值且不等于 \(x\)，则去下一个位置继续查找；如果没有值则 \(x\) 不存在。

    删除：按照查找的方式找 \(x\)，然后用 bool 数组标记 \(x\)。

    核心：find 函数（蹲坑法）

    如果 \(x\) 在哈希表中存在的话，find 函数会返回它存储的位置；

    如果 \(x\) 不存在的话，则会返回它应该存储的位置。

    #include <iostream>
#include <cstring>

using namespace std;

const int N = 200003, flag = 0x3f3f3f3f; // 100000 的 2-3 倍，且是质数；0x3f3f3f3f 大于 10^9

int h[N]; // h 储存的是每个单链表的头节点

int find(int x) {
    int k = (x % N + N) % N;

    while(h[k] != flag && h[k] != x) {
        k++;
        if (k == N) k = 0;
    }

    return k;
}

int main() {
    int n;
    scanf("%d", &n);

    memset(h, 0x3f, sizeof h); // 把所有的槽清空，-1 表示空指针

    while (n--) {
        char op[2];
        int x;
        scanf("%s%d", op, &x);

        int k = find(x);
        if (op[0] == 'I') h[k] = x;
        else {
            if (h[k] != flag) printf("Yes\n");
            else printf("No\n");
        }
    }

    return 0;
}

    字符串哈希方式¶

    字符串前缀哈希法

    1. 将字符串看成 p 进制的数

    2. 将 p 进制数转化成 10 进制数，例如 "ABCD" 可以看作 \((1234)_p\) $$ res = 1 \times p^3 + 2 \times p^2 + 3 \times p^1 + 4 \times p^0 $$

    3. 将结果 mod Q（结果可能比较大），将字符串映射到 \([0, Q-1]\)

    注：

    1. 不能将某一字符经过 hash 后映射成 0，如 \(hash(A) = 0\)，则 \(hash(AA)=0\)，会产生冲突。

    2. 假设人品足够好，不存在冲突，有一组经验值（99.99% 的情况下都没问题） $$ \begin{aligned} & p = 131 & Q = 2^{64} \ \text{or} \quad & p = 13331 & Q = 2^{64} \end{aligned} $$

    好处是可以利用前缀哈希算出所有子串的哈希值

q.empty(); // 返回队列是否为空
q.push(x); // 向队尾插入 x
q.front(); // 返回队头元素
q.back(); // 返回队尾元素
q.pop(); // 弹出队头元素

    queue, priority_queue 和 stack 都没有 clear 函数，如果想清空，直接重新新建一个就行。

    set, map, multiset, multimap¶

    基于平衡二叉树（红黑树）实现，动态维护有序序列。

    set, multiset¶

    set 中没有重复元素（重复会被忽略），multiset 中可以有重复元素

    头文件：

    除 size(), empty() 和 clear() 是 \(\mathcal{O}(1)\) 外，set 其他操作的时间复杂度都是 \(\mathcal{O}(\log n)\)。

    set, multiset 最核心的操作是 lower_bound() 和 upper_bound()（最小上界和最大下界）

    set<int> s;
multiset<int> ms;
s.size(), ms.size();
s.empty(), ms.empty();
s.clear(), ms.clear();
s.begin(), ms.begin(); // 支持 ++ -- 操作，返回前驱 O(logn)
s.end(), ms.end(); // 支持 ++ -- 操作，返回后继 O(logn)
s.insert(x), ms.insert(x); // 插入 x
s.find(x), ms.find(x); // 查找 x，如果不存在则返回 end 迭代器
s.count(x), ms.count(x); // 返回 set 中 x 的个数（0, 1）
s.erase(), ms.erase(); // 若输入是一个数 x，则删除所有 x，时间复杂度为 O(k + logn)，k 为 x 的个数；若输入是一个迭代器，则删除这个迭代器（multiset）
s.lower_bound(x), ms.lower_bound(x); // 返回大于等于 x 的最小的数的迭代器，若不存在则返回 end
s.upper_bound(x), ms.upper_bound(x); // 返回大于 x 的最小的数的迭代器，若不存在则返回 end

    map, multimap¶

    头文件：

    除 size(), empty() 和 clear() 是 \(\mathcal{O}(1)\) 外，map 其他操作的时间复杂度都是 \(\mathcal{O}(\log n)\)。

    map<string, int> mp;
mp.size();
mp.empty();
mp.clear();
mp.begin(), mp.end(); // 支持 ++ -- 操作，返回前驱和后继 O(logn)
mp.insert(x); // 用的不多，插入的数是一个 pair
mp.erase(x); // 用的较多，输入的参数是 pair 或迭代器
mp.find(x);
mp.lower_bound(x), mp.upper_bound(x);
// [] 可以像数组一样使用 map，但这样的时间复杂度是 O(logn)

    例子

    #include <iostream>
#include <map>

int main() {
    map<string, int> mp;

    mp["yxc"] = 1;

    cout << mp["yxc"] << endl;

    return 0;
}

    bitset¶

    压位

    bool

    例如储存 \(10000 \times 10000\) 的 bool 矩阵，正常储存需要 \(10^8 \approx 100 \text{MB}\) 空间，若使用 bitset，只需要 \(1/8\) 的空间（\(12 \text{MB}\)）。

    bitset<10000> s; // 定义长度为 10000 的 bitset
~, &, |, ^; // 可以将 bitset 看作一个整数，支持位运算
>>, <<;
==, !=;
// []，支持取出来某一位
s.count(); // 返回 bitset 中有多少个 1
s.any(); // 返回是否至少有一个 1
s.none(); // 返回是否全为 0
s.set(); // 把所有位置为 1
s.set(k, v); // 把第 k 位变成 v
s.reset(); // 把所有位置为 0
s.flip(); // 等价于 ~
s.flip(k); // 把第 k 位取反
