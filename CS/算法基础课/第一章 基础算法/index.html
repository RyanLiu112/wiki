    快速排序¶

    基本思想：分治

    时间复杂度：\(\mathcal{O}(n\log n)\)，注意快排的时间复杂度是平均而言的，因为快排划分出来的两个区间长度不一定相等，但长度的期望是n/2。层数的期望是nlogn，所以是2*n/2*nlogn。

    步骤

    • 确定分界点：\(q[l], q[(l+r)/2], q[r]\), 随机
    • 调整区间（重难点）：使得左边区间所有的数都小于等于x，右边区间所有的数都大于等于x
    • 递归处理左右两段

    快速排序是不稳定的，两个相同的数改变位置可能会导致这两个数排序结果不同。

    在任意时刻，i 左边的数都小于x，j 右边的数都大于x

    785.快速排序¶

    链接：https://www.acwing.com/problem/content/787/

    #include <iostream>

using namespace std;

const int N = 100010;

int n;
int q[N];

void quick_sort(int q[], int l, int r) {
    if (l >= r) return;

    int x = q[(l + r) >> 1], i = l - 1, j = r + 1;
    while (i < j) {
        do i++; while (q[i] < x);
        do j--; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }

    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", &q[i]);

    quick_sort(q, 0, n - 1);

    for (int i = 0; i < n; ++i) printf("%d ", q[i]);

    return 0;
}

    快排正宗写法¶

    #include <iostream>

using namespace std;

const int N = 100010;

int n;
int q[N];

void quick_sort(int q[], int left, int right) {
    if (left >= right) return;

    int pivot = q[left], i = left, j = right;
    while (i < j) {
        while (i < j && q[j] >= pivot) j--;
        while (i < j && q[i] <= pivot) i++;
        swap(q[i], q[j]);
    }
    swap(q[left], q[i]);
    quick_sort(q, left, i - 1);
    quick_sort(q, i + 1, right);
}

int main() {
    scanf("%d", &n);

    for (int i = 0; i < n; ++i) scanf("%d", &q[i]);

    quick_sort(q, 0, n - 1);

    for (int i = 0; i < n; ++i) printf("%d ", q[i]);

    return 0;
}

    786.第k个数¶

    链接：https://www.acwing.com/problem/content/788/

    #include <iostream>

using namespace std;

const int N = 100010;

int n;
int q[N];

void quick_sort(int q[], int l, int r) {
    if (l >= r) return;

    int x = q[(l + r) >> 1], i = l - 1, j = r + 1;

    while (i < j) {
        do i++; while (q[i] < x);
        do j--; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }

    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

    int k;

    quick_sort(q, 0, n - 1);

    printf("%d\n", q[k - 1]);

    return 0;
}

    归并排序¶

    基本思想：分治

    时间复杂度：\(\mathcal{O}(n\log n)\)，归并排序的时间复杂度比较稳定。

    步骤：

    • 确定分界点：mid = (l + r) / 2
    • 递归排序left和right
    • （重难点）归并，将两个有序的数组合成一个有序的数组（\(\mathcal{O}(n)\)，每个元素只会被比较一次）。

    归并排序是稳定的。

    787.归并排序¶

    链接：https://www.acwing.com/problem/content/789/

    #include <iostream>

using namespace std;

const int N = 100010;

int n;
int q[N], tmp[N]; // tmp 用于储存归并排序后的结果

void merge_sort(int q[], int l, int r) { // [l, r] 闭区间
    if (l >= r) return;

    int mid = (l + r) >> 1;

    merge_sort(q, l, mid); // 递归排序左边 [l, mid]
    merge_sort(q, mid + 1, r); // 递归排序右边 [mid + 1, r]

    int k = 0, i = l, j = mid + 1; // i 指向左边的起点，j 指向右边的起点
    while (i <= mid && j <= r) {
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    }
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];

    for (i = l, j = 0; i <= r; ++i, ++j) q[i] = tmp[j];
}

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; ++i) scanf("%d", &q[i]);

    merge_sort(q, 0, n - 1);

    for (int i = 0; i < n; ++i) printf("%d ", q[i]);

    return 0;
}

    二分¶

    • 整数二分（很蛋疼，会有很多边界问题，写不好会死循环）
    • 浮点数二分

    二分的本质¶

    二分的本质是边界，而不是单调性。

    如果有单调性，则一定可以二分；但可以二分的题目，不一定需要单调性。

    整数二分¶

    模板一（红色，找右边界）¶
    • 找中间点：mid = (l + r + 1) / 2
    • 检查mid是否满足
    • if (check(mid))：则mid在左边区间，则答案在[mid, r]，l = mid
    • else：则答案在[l, mid - 1]，r = mid - 1
    模板二（绿色，找左边界）¶
    • 找中间点：mid = (l + r) / 2
    • 检查mid是否满足
    • if (check(mid))：则mid在右边区间，则答案在[l, mid]，r = mid
    • else：则答案在[mid + 1, r]，l = mid + 1

    如何选择模板？¶

    先写check函数，根据check想如何更新区间，l = mid则用模板一（mid = (l + r + 1) / 2），r = mid则用模板二。

    模板一中mid = (l + r + 1) / 2是因为，假设mid = (l + r) / 2，当l = r-1时，mid=l，若check(mid)为true，则更新区间时l = mid = l，会死循环。

    789.数的范围¶

    链接：https://www.acwing.com/problem/content/791/

    模板一

    int binary_search() {
    int l = 0, r = n - 1;
    while (l < r) {
        int mid = (l + r + 1) >> 1;
        if (check(mid)) l = mid;
        else r = mid + 1;
    }

    return l;
}

    模板二

    int binary_search() {
    int l = 0, r = n - 1;
    while (l < r) {
        int mid = (l + r) >> 1;
        if (check(mid)) r = mid;
        else l = mid - 1;
    }

    return l;
}

    #include <iostream>

using namespace std;

const int N = 100010;

int n, m;
int q[N];

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; ++i) scanf("%d", &q[i]);

    while (m--) {
        int x;
        scanf("%d", &x);

        int l = 0, r = n - 1;
        while (l < r) {
            int mid = (l + r) >> 1;
            if (q[mid] >= x) r = mid;
            else l = mid + 1;
        }

        if (q[l] != x) printf("-1 -1\n");
        else {
            printf("%d ", l);

            int l = 0, r = n - 1;
            while (l < r) {
                int mid = (l + r + 1) >> 1;
                if (q[mid] <= x) l = mid;
                else r = mid - 1;
            }
            printf("%d\n", l);
        }
    }

    return 0;
}

    二分正宗写法¶

    左闭右闭区间¶
    int binary_search(vector<int>& nums, int target) {
    int left = 0, right = nums.size() - 1;
    while (left <= right) { // 根据闭区间定义，可以加等号
        int mid = (left + right) >> 1;
        if (nums[mid] > target) r = mid - 1;
        else if (nums[mid] < target) l = mid + 1;
        else return mid;
    }
    return -1;
}
    左闭右开区间¶
    int binary_search(vector<int>& nums, int target) {
    int left = 0, right = nums.size() - 1;
    while (left < right) { // 根据左闭右开区间定义，不能加等号
        int mid = (left + right) >> 1;
        if (nums[mid] > target) r = mid;
        else if (nums[mid] < target) l = mid + 1;
        else return mid;
    }
    return -1;
}

    790.数的三次方根¶

    链接：https://www.acwing.com/problem/content/792/

    #include <iostream>

using namespace std;

int main() {
    double n;
    scanf("%lf", &n);

    double l = -10000, r = 10000;
    while (r - l > 1e-8) { // 要求保留6位小数，所以至少要1e-8
        double mid = (l + r) / 2;
        if (mid * mid * mid >= n) r = mid;
        else l = mid;
    }

    printf("%.6lf\n", l);

    return 0;
}
